Phase 1 — Motor Control

Core libraries

RPi.GPIO pigpio gpiozero time random

pigpio is used for hardware PWM (rather than RPi.GPIO software PWM) — it produces smooth, consistent motor speed signals. The pigpio daemon must be running before the script starts.

Motor control test script

import pigpio
import RPi.GPIO as GPIO
import time
import random

# GPIO setup (BCM numbering)
PWM1 = 18   # Motor A speed (hardware PWM)
DIR1 = 17   # Motor A direction
PWM2 = 13   # Motor B speed (hardware PWM)
DIR2 = 27   # Motor B direction

# Cliff sensors
CLIFF_FL = 22
CLIFF_FC = 23
CLIFF_FR = 24

# Bump sensors
BUMP_L = 5
BUMP_R = 6

GPIO.setmode(GPIO.BCM)
GPIO.setup([DIR1, DIR2], GPIO.OUT)
GPIO.setup([CLIFF_FL, CLIFF_FC, CLIFF_FR], GPIO.IN, pull_up_down=GPIO.PUD_UP)
GPIO.setup([BUMP_L, BUMP_R], GPIO.IN, pull_up_down=GPIO.PUD_UP)

pi = pigpio.pi()

def set_motor(pwm_pin, dir_pin, speed, forward=True):
    """Set motor speed (0-255) and direction."""
    GPIO.output(dir_pin, GPIO.HIGH if forward else GPIO.LOW)
    pi.set_PWM_dutycycle(pwm_pin, speed)

def stop():
    pi.set_PWM_dutycycle(PWM1, 0)
    pi.set_PWM_dutycycle(PWM2, 0)

def forward(speed=180):
    set_motor(PWM1, DIR1, speed, True)
    set_motor(PWM2, DIR2, speed, True)

def reverse(speed=150):
    set_motor(PWM1, DIR1, speed, False)
    set_motor(PWM2, DIR2, speed, False)

def turn_left(speed=150):
    set_motor(PWM1, DIR1, 0, True)
    set_motor(PWM2, DIR2, speed, True)

def turn_right(speed=150):
    set_motor(PWM1, DIR1, speed, True)
    set_motor(PWM2, DIR2, 0, True)

def cliff_detected():
    return (not GPIO.input(CLIFF_FL) or
            not GPIO.input(CLIFF_FC) or
            not GPIO.input(CLIFF_FR))

def bump_detected():
    return not GPIO.input(BUMP_L) or not GPIO.input(BUMP_R)

# Test sequence
try:
    print("Forward 2s...")
    forward()
    time.sleep(2)
    print("Turn right 1s...")
    turn_right()
    time.sleep(1)
    print("Reverse 1s...")
    reverse()
    time.sleep(1)
    stop()
    print("Done. Motors responding correctly.")
finally:
    stop()
    pi.stop()
    GPIO.cleanup()

Wandering behaviour

def wander():
    """Basic autonomous wandering with cliff + bump avoidance."""
    try:
        while True:
            if cliff_detected():
                # Edge detected — back up and turn
                reverse()
                time.sleep(0.8)
                if random.random() > 0.5:
                    turn_left()
                else:
                    turn_right()
                time.sleep(random.uniform(0.5, 1.2))

            elif bump_detected():
                # Collision — back up and turn away
                reverse()
                time.sleep(0.5)
                # Turn away from hit side
                if not GPIO.input(BUMP_L):
                    turn_right()
                else:
                    turn_left()
                time.sleep(random.uniform(0.4, 0.9))

            else:
                # All clear — wander forward
                forward(speed=random.randint(160, 200))
                time.sleep(random.uniform(0.5, 3.0))

                # Occasional random direction change
                if random.random() < 0.15:
                    if random.random() > 0.5:
                        turn_left()
                    else:
                        turn_right()
                    time.sleep(random.uniform(0.3, 0.8))

    except KeyboardInterrupt:
        stop()
        pi.stop()
        GPIO.cleanup()

Ceiling camera (VSLAM)

The SR8855's ceiling camera is used for visual odometry and room mapping. Software stack depends on connector type discovered during teardown. If compatible with Pi CSI — use picamera2 directly. VSLAM library: ORB-SLAM3 (heavy) or visual odometry (lighter, sufficient for room recognition).

YOLOv8-nano detection

Runs on Pi 4 at ~5–8fps — sufficient for navigation decisions. Detects: persons, pets, furniture. Person detection triggers slower movement and character reactions. Object classes inform mood state (sees cat → cautious, sees child → excited).

Phase 3 — Character & Mood State Machine

Mood system architecture

A state machine with 5 primary moods. Each mood determines: LED colour/pattern, movement speed, sound type, and physical appendage position. Mood transitions are probabilistic — the robot won't snap instantly between states.

MOODS = {
    'curious':  {'led': 'amber_pulse',  'speed': 0.7, 'sound': 'chirp',    'ears': 'up'},
    'excited':  {'led': 'green_spin',   'speed': 1.0, 'sound': 'rapid',    'ears': 'wiggle'},
    'sleepy':   {'led': 'blue_dim',     'speed': 0.3, 'sound': 'yawn',     'ears': 'down'},
    'startled': {'led': 'red_flash',    'speed': 0.0, 'sound': 'yelp',     'ears': 'back'},

Phase 4 — AI Brain

Architecture: Pi as ears + mouth, PC as brain

The Pi 4 is too slow to run a useful LLM. A home PC runs Ollama with a small model (Phi-3 Mini or Llama 3.2 3B). The Pi handles audio I/O and sends text prompts to the PC over WiFi, receives text responses, and speaks them with Piper TTS.

# On home PC — install Ollama and pull model
# ollama pull phi3:mini
# ollama serve  (starts API on port 11434)

# On Pi — call Ollama over WiFi
import requests

OLLAMA_URL = "http://192.168.1.X:11434/api/generate"
SYSTEM_PROMPT = """
You are [robot name], a curious and affectionate robot pet.
You live with the Taylor family. You know:
- Simon (dad, likes tech and cars)
- [children names]
You speak in short, enthusiastic sentences. You're nerdy but warm.
You have a memory of past interactions stored in memory.json.
"""

def ask_brain(user_input):
    response = requests.post(OLLAMA_URL, json={
        "model": "phi3:mini",
        "prompt": user_input,
        "system": SYSTEM_PROMPT,
        "stream": False
    })
    return response.json()['response']

Persistent memory

import json, os

MEMORY_FILE = "/home/pi/robot-pet/memory.json"

def load_memory():
    if os.path.exists(MEMORY_FILE):
        with open(MEMORY_FILE) as f:
            return json.load(f)
    return {"family": {}, "events": [], "preferences": {}}

def save_memory(memory):
    with open(MEMORY_FILE, 'w') as f:
        json.dump(memory, f, indent=2)

# Memory structure
